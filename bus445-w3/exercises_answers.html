# Actually, let's just convert to factor
egg_data <- egg_data %>%
  mutate(
    Month = factor(Month, levels = unique(Month)),

# Boxplot of Cases by First.Week

2. Exercise 4.5.8. Skill: Interpreting Scatterplots

Question: Identify any other variables that might be related to Egg Case sales using scatterplots.

# Scatterplots for continuous variables
continuous_vars <- c("Egg.Pr", "Beef.Pr", "Pork.Pr", "Chicken.Pr", "Cereal.Pr")

# Pivot longer to plot all at once using facets
egg_data %>%
  select(Cases, all_of(continuous_vars)) %>%
  pivot_longer(cols = -Cases, names_to = "Variable", values_to = "Value") %>%
  ggplot(aes(x = Value, y = Cases)) +
  geom_point() +
  geom_smooth(method = "lm", se = FALSE, color = "blue", size = 0.5) +
  facet_wrap(~ Variable, scales = "free_x") +
  labs(title = "Scatterplots of Cases vs Price Variables",
       y = "Cases sold") +
  theme_minimal()
